  drop table survey;
  create table survey(
    id int,
  
  --Group: Location --
  city varchar(50),
  areacode varchar(50),
  
  --Group: Presentations --
  present varchar(50),
  
  --Group: Meetings --
  attendances varchar(50),
  often varchar(50),
  day varchar(50),
  time varchar(50),
  earliest varchar(50),
  latest varchar(50),
  
  --Group: Participation --
  volunteer varchar(50),
  
  --Group: Knowledge --
  understand_OO varchar(50),
  methodology_OO varchar(50),
  learn_OO varchar(50),
  catagory varchar(50),
  language varchar(50),
  programmer_months varchar(50),
  java_months varchar(50),
  java_skill varchar(50),
  opensource_use varchar(50),
  opensource_contrib varchar(50),
  
  --Group: Development Environment --
  ide varchar(50),
  server varchar(50),
  os varchar(50),
  
  --Group: Employment --
  position varchar(50),
  seeking varchar(50),
  available varchar(50),
  hiring varchar(50),
  hiring_type varchar(50),
  contract_months varchar(50),
  
    primary key(id)
  );
